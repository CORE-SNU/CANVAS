cmake_minimum_required(VERSION 2.8.3)
project(gazebo_rossim)
add_compile_options(-std=c++17)

find_package(catkin REQUIRED COMPONENTS
  gazebo_ros
  gazebo_plugins
  geometry_msgs
  nav_msgs
)

find_package(gazebo REQUIRED)

catkin_package(
  INCLUDE_DIRS 
    include
  LIBRARIES 
    gazebo_ros_actor_command_1
    gazebo_ros_actor_command_2
    gazebo_ros_actor_command_3
    gazebo_ros_actor_command_4
  CATKIN_DEPENDS
    gazebo_ros
    gazebo_plugins
    geometry_msgs
    nav_msgs
)

include_directories(
  include
  ${catkin_INCLUDE_DIRS}
  ${GAZEBO_INCLUDE_DIRS}
)

link_directories(${GAZEBO_LIBRARY_DIRS})
list(APPEND CMAKE_CXX_FLAGS "${GAZEBO_CXX_FLAGS}")

catkin_install_python(PROGRAMS scripts/odom_pub_actor.py DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION})

add_library(gazebo_ros_actor_command_1 src/gazebo_ros_actor_command_1.cpp)
target_link_libraries(gazebo_ros_actor_command_1 ${catkin_LIBRARIES} ${GAZEBO_LIBRARIES})

add_library(gazebo_ros_actor_command_2 src/gazebo_ros_actor_command_2.cpp)
target_link_libraries(gazebo_ros_actor_command_2 ${catkin_LIBRARIES} ${GAZEBO_LIBRARIES})

add_library(gazebo_ros_actor_command_3 src/gazebo_ros_actor_command_3.cpp)
target_link_libraries(gazebo_ros_actor_command_3 ${catkin_LIBRARIES} ${GAZEBO_LIBRARIES})

add_library(gazebo_ros_actor_command_4 src/gazebo_ros_actor_command_4.cpp)
target_link_libraries(gazebo_ros_actor_command_4 ${catkin_LIBRARIES} ${GAZEBO_LIBRARIES})
